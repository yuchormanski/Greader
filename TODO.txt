  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && 
					  request.resource.contentType == 'video/mp4' && 
					  // request.resource.contentType.matches('video/*')  -за всички видове видео файлове
					  request.resource.size < 10 * 1000 * 1000;
    }
  }
